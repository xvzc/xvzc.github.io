#include <bits/stdc++.h>

using namespace std;

vector<int> solution(vector<int> array, vector<vector<int>> commands) {
    vector<int> answer;
    for (auto v : commands) {
        int i = v[0];
        int j = v[1];
        int k = v[2];
        
        vector<int> sub(array.begin()+i-1, array.begin()+j);
        
        priority_queue<int> pq;
        for (auto n : sub) {
            if (pq.size() < k) {
                pq.push(n);
                continue;
            }
            
            if (pq.top() > n) {
                pq.pop();
                pq.push(n);
            }
        }
        
        answer.push_back(pq.top());
    }
    return answer;
}
